348     public CommitUnpack getCommits(String user, String branchName) throws IOException {
349         List<CommitHttpRequest> commits = new ArrayList<>();
350 
351         int page = 1;
352         while (true) {
353             Request request = new Request.Builder()
354                     .addHeader("Authorization", "Bearer " + GITHUB_OAUTH)
355                     .url(String.format(this.url + "/commits?" + "&author=" + user + "&sha=" + branchName + "&page=%d", page++)).build();
356             try (Response response = client.newCall(request).execute()) {
357                 try {
358                     var cm = this.mapper.readValue(Objects.requireNonNull(response.body()).string(), CommitHttpRequest[].class);
359                     if (cm.length == 0) {
360                         break;
361                     }
362 
363                     commits.addAll(Arrays.asList(cm));
364                 } catch (MismatchedInputException e) {
365                     break;
366                 }
367             }
368         }
369         return new CommitUnpack(user, commits);
370     }
